\section{Styles}

\subsection{No border}
Corners without a border line. You can still use a single line with them, but
that might look better for some than for others.

\par\vspace{1em}%

\newcounter{loopctr}%
\forloop{loopctr}{1}{\value{loopctr} < 17}{%
  \hfill%
  \ifnum\value{loopctr}<10%
    \def\style{style-none-0\theloopctr}%
  \else%
    \def\style{style-none-\theloopctr}%
  \fi%
  \begin{minipage}{0.24\textwidth}%
    \framedbox[width=4cm, height=3cm, corner=\style, cornersize=1cm, mode=full,
    lines=0]{\style, \\
      mode=full, \\
      lines=0}%
  \end{minipage}%
  \hfill%
  \begin{minipage}{0.24\textwidth}%
    \framedbox[width=4cm, height=3cm, corner=\style, cornersize=1cm, mode=full,
    lines=1]{\style, \\
      mode=full, \\
      lines=1}%
  \end{minipage}%
  \begin{minipage}{0.24\textwidth}%
    \framedbox[width=4cm, height=3cm, corner=\style, cornersize=1cm, mode=diagonal,
    lines=1]{\style, \\
      mode=diagonal, \\
      lines=1}%
  \end{minipage}%
  \begin{minipage}{0.24\textwidth}%
    \framedbox[width=4cm, height=3cm, corner=\style, cornersize=1cm, mode=antidiagonal,
    lines=0]{\style, \\
      mode=antidiagonal, \\
      lines=0}%
  \end{minipage}%
  \par\vspace{1em}%
}
